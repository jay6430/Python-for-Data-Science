from lxml import objectify
import pandas as pd

xml = objectify.parse(open('/content/drive/My Drive/Site sharing/PSD Resources/XMLData.xml'))
#parsing the data file using the objectify.parse() method.
#parsing is to analyze the data

root = xml.getroot()
#Every XML document must contain a root node, which is <MyDataset> in this case.
#The root node encapsulates the rest of the content, and every node under it is a child.
#you must obtain access to the root node using the getroot() method.



df = pd.DataFrame(columns=('Number', 'String', 'Boolean'))
#This create an empty DataFrame object that contains the correct column names for each record entry: Number, String, and Boolean. 


#The for loop fills the DataFrame with the four records from the XML file (each in a <Record> node).
for i in range(0,4):
    
    obj = root.getchildren()[i].getchildren() 
    #subdata ie: childern of the root node
    #The for loop fills the DataFrame with the four records from the XML file (each in a <Record> node).
    
    row = dict(zip(['Number', 'String', 'Boolean'],
                   [obj[0].text, obj[1].text,obj[2].text]))
    # These children are loaded into a dictionary object in which the keys are Number, String, and Boolean to match the Data.
    
    row_s = pd.Series(row)
    #series created
  
    row_s.name = i
    # giving name to the series

    df = df.append(row_s)
    # Append all the rows in the dataframe which are iterated in the loop

print(df)
print()




print("\n\n\n")
print(obj) #childrens
print()
print(row)   #dictionary
print()
print(row_s)  #series
print()
print(row_s.name)  #name of the series

  4. variablename.shape -Gives the shape of the image,, usually the dimensions
  5. variablename.size -Gives the size of the image usually the multiplication of the values in shape
  6. plt.imshow(variablename) -Display array data as an image
  7. x = ('Kushal', 'Aadarsh', 'Jay')

    -enumerate() function takes a collection/data (e.g. a tuple) and returns it as an enumerate object.
    -In simple language enumerate will attach numbering starting from 0 to n for the data provided.

    -A semi-open range is one which includes the first element(thus square bracket), but excludes the last one (thus round bracket).
